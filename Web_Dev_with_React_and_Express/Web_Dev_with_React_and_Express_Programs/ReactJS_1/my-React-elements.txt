//This function sets the time and date and renders them to their respectful tags
function timeDate() {
  var newDate = new Date();
  // [lines 14-17] get the time
  var hrs = newDate.getHours();
  var mins = newDate.getMinutes();
  var secs = newDate.getSeconds();
  var time =
    ("0" + hrs).slice(-2) +
    ":" +
    ("0" + mins).slice(-2) +
    ":" +
    ("0" + secs).slice(-2);
  //[line 24-27] gets the date
  var day = newDate.getDate();
  var month = newDate.getMonth();
  var year = newDate.getFullYear();
  var mnth = parseInt(month) + 1;
  var date = ("0" + day).slice(-2) + "/" + mnth + "/" + year;
  //[lines 30-31] renders the time and date
  ReactDOM.render(time, document.getElementById("time"));
  ReactDOM.render(date, document.getElementById("date"));
}
//a setInterval for every 1 second
setInterval(timeDate, 1000);
